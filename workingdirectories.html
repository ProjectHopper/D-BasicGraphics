%==============================================================================================% \begin{frame} \frametitle{1.12 Working directories} \large \begin{itemize} \item You can change your working directly by using the appropriate options on the File menu. \item To determine the current working directory, you can use the \texttt{getwd()} command. \item To change the working directory , we would use the \texttt{setwd()} command. \item This is quite important as objects will be imported and exported to and from the specified directory. \end{itemize} \end{frame} %==============================================================================================% \begin{frame} \begin{figure} \centering \includegraphics[width=1.2\linewidth]{images/workingdir} \end{figure} \end{frame}
